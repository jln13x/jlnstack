# Installation

<Tabs items={["npm", "pnpm", "yarn", "bun"]}>
  <Tab value="npm">
    ```bash
    npm install @jlnstack/filter
    ```
  </Tab>
  <Tab value="pnpm">
    ```bash
    pnpm add @jlnstack/filter
    ```
  </Tab>
  <Tab value="yarn">
    ```bash
    yarn add @jlnstack/filter
    ```
  </Tab>
  <Tab value="bun">
    ```bash
    bun add @jlnstack/filter
    ```
  </Tab>
</Tabs>

## Entry Points

```ts
// Core filter definitions and expression helpers (no React dependency)
import { 
  defineFilters, 
  createFilter, 
  and, 
  or, 
  condition 
} from "@jlnstack/filter"

// React hooks and components
import { useFilter, FilterProvider, createFilterHooks } from "@jlnstack/filter/react"

// Drizzle ORM integration
import { toWhere } from "@jlnstack/filter/drizzle"

// Devtools for visualization
import { generateMermaid, getMermaidLiveUrl } from "@jlnstack/filter/devtools"
```
